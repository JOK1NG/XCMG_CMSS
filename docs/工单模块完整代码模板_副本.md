# CMMS 工单模块完整代码模板
## WorkOrderController + DTO + VO + Entity + Mapper + Mapper.xml + Service + ServiceImpl

## 1. 说明

本文档提供一套可直接落地的 SpringBoot 工单模块代码模板，适合作为整个 CMMS 后端的核心样板模块。

覆盖内容：

- Entity
- DTO
- VO
- Mapper
- Mapper.xml
- Service
- ServiceImpl
- Controller
- Enum
- 关键业务规则说明

技术假设：

- Spring Boot 3.x
- MyBatis
- MySQL
- RESTful API
- 统一返回结构 `Result`
- 全局异常 `BusinessException`

package com.example.cmms.modules.workorder.enumtype;

public enum WorkOrderStatusEnum {

    TO_ASSIGN(1, "待派工"),
    ASSIGNED(2, "已派工"),
    REPAIRING(3, "维修中"),
    TO_CHECK(4, "待验收"),
    FINISHED(5, "已完成"),
    CLOSED(6, "已关闭");

    private final Integer value;
    private final String label;

    WorkOrderStatusEnum(Integer value, String label) {
        this.value = value;
        this.label = label;
    }

    public Integer getValue() {
        return value;
    }

    public String getLabel() {
        return label;
    }

    public static String getLabelByValue(Integer value) {
        for (WorkOrderStatusEnum item : values()) {
            if (item.value.equals(value)) {
                return item.label;
            }
        }
        return "";
    }
}

package com.example.cmms.modules.workorder.service.impl;

import com.example.cmms.modules.workorder.entity.WorkOrderLog;
import com.example.cmms.modules.workorder.mapper.WorkOrderLogMapper;
import com.example.cmms.modules.workorder.service.WorkOrderLogService;
import org.springframework.stereotype.Service;

import java.time.LocalDateTime;

@Service
public class WorkOrderLogServiceImpl implements WorkOrderLogService {

    private final WorkOrderLogMapper workOrderLogMapper;

    public WorkOrderLogServiceImpl(WorkOrderLogMapper workOrderLogMapper) {
        this.workOrderLogMapper = workOrderLogMapper;
    }

    @Override
    public void addLog(Long workOrderId, String actionType, String actionLabel, Long operatorId, String operatorName, String remark) {
        WorkOrderLog log = new WorkOrderLog();
        log.setWorkOrderId(workOrderId);
        log.setActionType(actionType);
        log.setActionLabel(actionLabel);
        log.setOperatorId(operatorId);
        log.setOperatorName(operatorName);
        log.setRemark(remark);
        log.setActionTime(LocalDateTime.now());
        workOrderLogMapper.insert(log);
    }
}

import com.example.cmms.modules.workorder.vo.WorkOrderLogVo;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;

@Service
public class WorkOrderServiceImpl implements WorkOrderService {

    private final WorkOrderMapper workOrderMapper;
    private final WorkOrderLogMapper workOrderLogMapper;
    private final WorkOrderLogService workOrderLogService;

    public WorkOrderServiceImpl(WorkOrderMapper workOrderMapper,
                                WorkOrderLogMapper workOrderLogMapper,
                                WorkOrderLogService workOrderLogService) {
        this.workOrderMapper = workOrderMapper;
        this.workOrderLogMapper = workOrderLogMapper;
        this.workOrderLogService = workOrderLogService;
    }

    @Override
    public PageResult<WorkOrderListVo> page(WorkOrderQueryDto dto) {
        List<WorkOrderListVo> list = workOrderMapper.selectPage(dto);
        Long total = workOrderMapper.selectPageCount(dto);
        return new PageResult<>(list, total, dto.getPageNum(), dto.getPageSize());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long report(WorkOrderReportDto dto) {
        WorkOrder workOrder = new WorkOrder();
        workOrder.setWorkOrderCode(generateWorkOrderCode());
        workOrder.setEquipmentId(dto.getEquipmentId());
        workOrder.setFaultDesc(dto.getFaultDesc());
        workOrder.setFaultLevel(dto.getFaultLevel());
        workOrder.setReporterId(dto.getReporterId());
        workOrder.setContactPhone(dto.getContactPhone());
        workOrder.setStatus(WorkOrderStatusEnum.TO_ASSIGN.getValue());

        workOrderMapper.insert(workOrder);

        workOrderLogService.addLog(
                workOrder.getId(),
                "REPORT",
                "发起报修",
                dto.getReporterId(),
                "系统用户",
                "提交报修"
        );
        return workOrder.getId();
    }

    @Override
    public WorkOrderDetailVo detail(Long id) {
        WorkOrderDetailVo detailVo = workOrderMapper.selectDetailById(id);
        if (detailVo == null) {
            throw new BusinessException("工单不存在");
        }
        List<WorkOrderLogVo> logList = workOrderLogMapper.selectByWorkOrderId(id);
        detailVo.setLogList(logList);
        return detailVo;
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void assign(Long id, WorkOrderAssignDto dto) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (!WorkOrderStatusEnum.TO_ASSIGN.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("当前状态不允许派工");
        }
        workOrder.setAssigneeId(dto.getAssigneeId());
        workOrder.setAssignUserId(1L);
        workOrder.setAssignTime(LocalDateTime.now());
        workOrder.setStatus(WorkOrderStatusEnum.ASSIGNED.getValue());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "ASSIGN", "派工", 1L, "系统管理员", dto.getAssignRemark());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void accept(Long id) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (!WorkOrderStatusEnum.ASSIGNED.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("当前状态不允许接单");
        }
        workOrder.setAcceptTime(LocalDateTime.now());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "ACCEPT", "接单", workOrder.getAssigneeId(), "维修人员", "已接单");
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void start(Long id, WorkOrderStartDto dto) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (!WorkOrderStatusEnum.ASSIGNED.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("当前状态不允许开始维修");
        }
        workOrder.setStartTime(LocalDateTime.now());
        workOrder.setStatus(WorkOrderStatusEnum.REPAIRING.getValue());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "START", "开始维修", workOrder.getAssigneeId(), "维修人员", dto.getStartRemark());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void finish(Long id, WorkOrderFinishDto dto) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (!WorkOrderStatusEnum.REPAIRING.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("当前状态不允许提交完工");
        }
        workOrder.setRepairRemark(dto.getRepairRemark());
        workOrder.setFinishTime(LocalDateTime.now());
        workOrder.setStatus(WorkOrderStatusEnum.TO_CHECK.getValue());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "FINISH", "提交完工", workOrder.getAssigneeId(), "维修人员", dto.getRepairRemark());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void check(Long id, WorkOrderCheckDto dto) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (!WorkOrderStatusEnum.TO_CHECK.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("当前状态不允许验收");
        }
        workOrder.setCheckResult(dto.getCheckResult());
        workOrder.setCheckRemark(dto.getCheckRemark());
        workOrder.setCheckTime(LocalDateTime.now());
        workOrder.setStatus(WorkOrderStatusEnum.FINISHED.getValue());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "CHECK", "验收", 1L, "验收人员", dto.getCheckRemark());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void close(Long id, WorkOrderCloseDto dto) {
        WorkOrder workOrder = getByIdOrThrow(id);
        if (WorkOrderStatusEnum.CLOSED.getValue().equals(workOrder.getStatus())) {
            throw new BusinessException("工单已关闭");
        }
        workOrder.setCloseRemark(dto.getCloseRemark());
        workOrder.setStatus(WorkOrderStatusEnum.CLOSED.getValue());
        workOrderMapper.updateById(workOrder);

        workOrderLogService.addLog(id, "CLOSE", "关闭工单", 1L, "系统管理员", dto.getCloseRemark());
    }

    private WorkOrder getByIdOrThrow(Long id) {
        WorkOrder workOrder = workOrderMapper.selectById(id);
        if (workOrder == null) {
            throw new BusinessException("工单不存在");
        }
        return workOrder;
    }

    private String generateWorkOrderCode() {
        return "WO-" + System.currentTimeMillis();
    }
}

## 12. 这一版模板还需要你后续补的点

这套模板已经能作为骨架直接开写，但还有 4 类内容建议继续补：

### 12.1 用户真实信息获取
现在示例里像 `operatorId = 1L`、`operatorName = "系统管理员"` 是占位写法。  
正式项目里应从登录上下文中取当前用户。

### 12.2 备件联动
`finish()` 里还没有真正接入：
- 备件出库
- 库存扣减
- 出库记录落表

### 12.3 详情联表信息
现在 `selectDetailById` 里只联了设备表。  
正式项目还应联：
- 报修人
- 派工人
- 维修人

### 12.4 状态 label 转换
当前 `statusLabel`、`faultLevelLabel` 可以在：
- XML 联 CASE WHEN
- Service 中用 Enum 转换
二选一统一处理。
